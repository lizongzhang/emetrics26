library(tidyverse)
library(gt)
library(patchwork)
library(ggrepel)

# ── 读取数据 ──────────────────────────────────────────────
library(readxl)
raw <- read_excel("df_24hr.xlsx")

# ── 清洗与整理 ────────────────────────────────────────────
df <- raw |>
  rename(
    姓名       = 姓名,
    学号       = 学号,
    班级       = 所属行政班,
    开课次数   = 开课次数,
    签到次数   = 签到次数,
    作业项数   = 作业和测验项数,
    作业提交数 = 作业和测验提交项数
  ) |>
  mutate(
    # 原始数据为 0–1，乘 100 转为百分制
    签到率     = 签到率     * 100,
    作业提交率 = 作业提交率 * 100,
    班级简称 = case_when(
      str_detect(班级, "1$") ~ "人力资源1班",
      str_detect(班级, "2$") ~ "人力资源2班",
      TRUE ~ 班级
    ),
    综合参与度 = (签到率 + 作业提交率) / 2,
    预警等级 = case_when(
      综合参与度 < 50 ~ "严重预警（<50%）",
      综合参与度 < 70 ~ "中度预警（50–70%）",
      综合参与度 < 85 ~ "轻度预警（70–85%）",
      TRUE            ~ "正常（≥85%）"
    ),
    预警等级 = factor(预警等级, levels = c(
      "严重预警（<50%）", "中度预警（50–70%）",
      "轻度预警（70–85%）", "正常（≥85%）"
    ))
  )

# 颜色方案（马卡龙色）
col_warning <- c(
  "严重预警（<50%）"   = "#F4A7B9",   # 玫瑰粉
  "中度预警（50–70%）" = "#FDBF8B",   # 蜜桃橙
  "轻度预警（70–85%）" = "#FFF0A0",   # 柠檬黄
  "正常（≥85%）"       = "#A8D8B9"    # 薄荷绿
)

# gt 表格通用样式函数
gt_style <- function(gt_tbl) {
  gt_tbl |>
    tab_options(
      table.width                        = pct(100),
      table.border.top.style             = "solid",
      table.border.top.width             = px(2),
      table.border.top.color             = "#4472C4",
      table.border.bottom.style          = "solid",
      table.border.bottom.width          = px(2),
      table.border.bottom.color          = "#4472C4",
      column_labels.border.bottom.width  = px(2),
      column_labels.border.bottom.color  = "#4472C4",
      data_row.padding                   = px(8),
      table.font.size                    = px(13)
    ) |>
    tab_style(
      style = list(
        cell_fill(color = "#4472C4"),
        cell_text(color = "white", weight = "bold", size = px(13))
      ),
      locations = cells_column_labels()
    )
}

# 预警表格辅助函数
make_warn_table <- function(data, header_color, row_color, title_text) {
  if (nrow(data) == 0) {
    cat("✅ 无此级别预警学生。")
    return(invisible(NULL))
  }
  data |>
    select(学号, 班级简称, 签到率, 作业提交率, 综合参与度) |>
    arrange(综合参与度) |>
    gt() |>
    tab_header(title = title_text) |>
    cols_label(
      学号       = "学号",
      班级简称   = "班级",
      签到率     = "签到率",
      作业提交率 = "作业提交率",
      综合参与度 = "综合参与度"
    ) |>
    fmt_number(
      columns  = c(签到率, 作业提交率, 综合参与度),
      decimals = 1,
      pattern  = "{x}%"
    ) |>
    cols_align(align = "center") |>
    cols_align(align = "left", columns = 学号) |>
    tab_style(
      style = list(
        cell_fill(color = header_color),
        cell_text(color = "white", weight = "bold", size = px(13))
      ),
      locations = cells_column_labels()
    ) |>
    tab_style(
      style = cell_fill(color = row_color),
      locations = cells_body()
    ) |>
    tab_options(
      table.width                       = pct(100),
      table.border.top.style            = "solid",
      table.border.top.width            = px(2),
      table.border.top.color            = header_color,
      table.border.bottom.style         = "solid",
      table.border.bottom.width         = px(2),
      table.border.bottom.color         = header_color,
      column_labels.border.bottom.width = px(2),
      column_labels.border.bottom.color = header_color,
      data_row.padding                  = px(8),
      table.font.size                   = px(13)
    )
}

6. 理论课与实验课参与情况对比

代码 
df2 <- df |>
  mutate(
    理论课签到率 = round(理论课学生签到次数 / 理论课班级开课数 * 100, 1),
    实验课签到率 = round(实验课学生签到次数 / 实验课班级开课数 * 100, 1)
  ) |>
  filter(!is.na(理论课签到率), !is.na(实验课签到率))

ggplot(df2, aes(x = 理论课签到率, y = 实验课签到率, color = 预警等级)) +
  geom_abline(slope = 1, intercept = 0,
              linetype = "dashed", color = "gray50") +
  geom_point(aes(shape = 班级简称), size = 3, alpha = 0.85) +
  geom_text_repel(
    data = filter(df2, abs(理论课签到率 - 实验课签到率) > 20 |
                       综合参与度 < 60),
    aes(label = 学号), size = 2.8, show.legend = FALSE,
    max.overlaps = 15, box.padding = 0.4
  ) +
  scale_color_manual(values = col_warning) +
  scale_x_continuous(labels = \(x) paste0(x, "%"), limits = c(0, 115)) +
  scale_y_continuous(labels = \(x) paste0(x, "%"), limits = c(0, 115)) +
  labs(
    title    = "理论课签到率 vs. 实验课签到率",
    subtitle = "虚线为理论课 = 实验课基准线；偏离较大或预警学生已标注学号",
    x = "理论课签到率", y = "实验课签到率",
    color = "预警等级", shape = "班级"
  ) +
  theme_bw(base_size = 13) +
  theme(legend.position = "right")
